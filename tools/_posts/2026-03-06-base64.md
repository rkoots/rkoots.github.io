    <h2>What is Base64 Encoding?</h2>
    <p>Base64 is a binary-to-text encoding scheme that converts binary data into a string of ASCII characters using a 64-character alphabet (A–Z, a–z, 0–9, +, /). It's widely used to embed binary data — such as images, audio, or files — inside text-based formats like JSON, XML, HTML, email (MIME), and data URIs.</p>
    <h3>When to Use Base64</h3>
    <ul>
      <li><strong>Embedding images in HTML/CSS</strong> – Use data URI format: <code>data:image/png;base64,...</code></li>
      <li><strong>API authentication</strong> – HTTP Basic Auth sends credentials as Base64-encoded "user:password"</li>
      <li><strong>Email attachments</strong> – MIME standard requires binary attachments to be Base64-encoded</li>
      <li><strong>JWT tokens</strong> – The header and payload of JSON Web Tokens are Base64URL-encoded</li>
      <li><strong>Storing binary in JSON</strong> – JSON doesn't support binary; Base64 is the standard workaround</li>
    </ul>
    <h3>Base64 vs Base64URL</h3>
    <p>Standard Base64 uses '+' and '/' characters which have special meaning in URLs. Base64URL replaces '+' with '-' and '/' with '_', and removes padding '=' characters. JWT tokens use Base64URL encoding. ENCODARA uses standard Base64 encoding.</p>
    <h3>Size Overhead</h3>
    <p>Base64-encoded data is approximately 33% larger than the original binary data (every 3 bytes become 4 characters). This is an acceptable tradeoff for the interoperability benefits in text-based protocols.</p>

<script>
let currentMode = 'encode';
const EXAMPLES = {
  hello: {text:'Hello, World! 🌍', mode:'encode'},
  json: {text:'{"name":"ENCODARA","version":1,"active":true}', mode:'encode'},
  url: {text:'https://rkoots.github.io/tools/base64/', mode:'encode'},
  decode: {text:'SGVsbG8sIFdvcmxkISAg8J+MjQ==', mode:'decode'}
};

function loadExample(key) {
  const ex = EXAMPLES[key];
  setMode(ex.mode);
  document.getElementById('left-input').value = ex.text;
  process();
}

function setMode(mode) {
  currentMode = mode;
  document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('btn-'+mode).classList.add('active');
  const left = document.getElementById('left-input');
  const right = document.getElementById('right-output');
  if (mode === 'encode') {
    document.getElementById('left-label').textContent = 'Plain Text';
    document.getElementById('right-label').textContent = 'Base64 Output';
    left.placeholder = 'Enter text to encode...';
  } else if (mode === 'decode') {
    document.getElementById('left-label').textContent = 'Base64 Input';
    document.getElementById('right-label').textContent = 'Decoded Text';
    left.placeholder = 'Paste Base64 string to decode...';
  } else {
    document.getElementById('left-label').textContent = 'Input (auto-detected)';
    document.getElementById('right-label').textContent = 'Output';
    left.placeholder = 'Enter text or Base64...';
  }
  if (left.value) process();
}

function isBase64(str) {
  try { return btoa(atob(str)) === str; } catch(e) { return false; }
}

function process() {
  const input = document.getElementById('left-input').value;
  const right = document.getElementById('right-output');
  document.getElementById('left-count').textContent = input.length + ' characters';
  if (!input.trim()) {
    right.value = '';
    document.getElementById('right-count').textContent = '0 characters';
    setStatus('info', 'Enter text to encode or Base64 to decode');
    right.classList.remove('error');
    return;
  }
  let mode = currentMode;
  if (mode === 'auto') mode = isBase64(input.trim()) ? 'decode' : 'encode';
  try {
    let result;
    if (mode === 'encode') {
      result = btoa(unescape(encodeURIComponent(input)));
      setStatus('ok', `✅ Encoded — ${input.length} chars → ${result.length} chars (${Math.round(result.length/input.length*100-100)}% overhead)`);
    } else {
      result = decodeURIComponent(escape(atob(input.trim())));
      setStatus('ok', `✅ Decoded — ${input.length} chars → ${result.length} chars`);
    }
    right.value = result;
    right.classList.remove('error');
    document.getElementById('right-count').textContent = result.length + ' characters';
  } catch(e) {
    right.value = '';
    right.classList.add('error');
    setStatus('err', '❌ Invalid Base64: ' + e.message);
    document.getElementById('right-count').textContent = '0 characters';
  }
}

function setStatus(type, msg) {
  const el = document.getElementById('status');
  el.className = 'status-strip ' + {ok:'ok', err:'err', info:'info'}[type];
  el.textContent = msg;
}

function swapPanels() {
  const left = document.getElementById('left-input');
  const right = document.getElementById('right-output');
  const temp = left.value;
  left.value = right.value;
  setMode(currentMode === 'encode' ? 'decode' : 'encode');
  process();
}

function encodeFile(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    const b64 = e.target.result.split(',')[1];
    document.getElementById('left-input').value = `[File: ${file.name}]\ndata:${file.type};base64,`;
    document.getElementById('right-output').value = b64;
    document.getElementById('right-count').textContent = b64.length + ' characters';
    setStatus('ok', `✅ File encoded: ${file.name} (${(file.size/1024).toFixed(1)} KB)`);
  };
  reader.readAsDataURL(file);
}

function copyRight() { navigator.clipboard.writeText(document.getElementById('right-output').value).then(()=>alert('Copied!')); }
function clearAll() { document.getElementById('left-input').value=''; document.getElementById('right-output').value=''; setStatus('info','Enter text to encode or Base64 to decode'); document.getElementById('left-count').textContent='0 characters'; document.getElementById('right-count').textContent='0 characters'; }
function toggleTheme() { const d=document.documentElement; d.setAttribute('data-theme',d.getAttribute('data-theme')==='dark'?'light':'dark'); }
function toggleFaq(el) { const a=el.nextElementSibling; const s=el.querySelector('span'); a.classList.toggle('hidden'); s.textContent=a.classList.contains('hidden')?'+':'−'; }
