<script>
let myIP = null;

async function fetchMyIP() {
  try {
    const res = await fetch('https://api.ipify.org?format=json');
    const data = await res.json();
    myIP = data.ip;
    document.getElementById('myIpDisplay').textContent = myIP;
  } catch(e) {
    document.getElementById('myIpDisplay').textContent = 'Unavailable';
  }
}

function copyMyIP() {
  if (myIP) navigator.clipboard.writeText(myIP).then(() => alert('IP copied: ' + myIP));
}

async function lookupMyIP() {
  document.getElementById('ipInput').value = '';
  await lookupIP();
}

async function lookupIP() {
  const ip = document.getElementById('ipInput').value.trim();
  const section = document.getElementById('resultSection');
  const content = document.getElementById('resultContent');
  section.style.display = 'block';
  content.innerHTML = '<div class="loader">🔍 Fetching IP data…</div>';

  try {
    const endpoint = ip ? `https://ipapi.co/${ip}/json/` : 'https://ipapi.co/json/';
    const res = await fetch(endpoint);
    if (!res.ok) throw new Error('API error ' + res.status);
    const d = await res.json();
    if (d.error) throw new Error(d.reason || 'Invalid IP address');

    const lat = d.latitude, lon = d.longitude;
    const mapUrl = lat && lon ? `https://www.openstreetmap.org/?mlat=${lat}&mlon=${lon}&zoom=10` : null;

    content.innerHTML = `
      <h2 style="font-size:1.1rem;color:var(--primary-dark);margin-bottom:14px;font-weight:700;">
        Results for: <span style="font-family:Consolas,monospace;">${d.ip}</span>
      </h2>
      <div class="result-grid">
        <div class="result-item"><div class="result-label">IP Address</div><div class="result-value mono">${d.ip || '—'}</div></div>
        <div class="result-item"><div class="result-label">Country</div><div class="result-value"><span class="flag">${getFlagEmoji(d.country_code)}</span>${d.country_name || '—'} (${d.country_code || '—'})</div></div>
        <div class="result-item"><div class="result-label">Region / State</div><div class="result-value">${d.region || '—'}</div></div>
        <div class="result-item"><div class="result-label">City</div><div class="result-value">${d.city || '—'}</div></div>
        <div class="result-item"><div class="result-label">Postal Code</div><div class="result-value">${d.postal || '—'}</div></div>
        <div class="result-item"><div class="result-label">Latitude / Longitude</div><div class="result-value mono">${lat || '—'}, ${lon || '—'}</div></div>
        <div class="result-item"><div class="result-label">Timezone</div><div class="result-value">${d.timezone || '—'}</div></div>
        <div class="result-item"><div class="result-label">UTC Offset</div><div class="result-value">${d.utc_offset ? 'UTC ' + d.utc_offset : '—'}</div></div>
        <div class="result-item"><div class="result-label">ISP / Organization</div><div class="result-value">${d.org || '—'}</div></div>
        <div class="result-item"><div class="result-label">ASN</div><div class="result-value mono">${d.asn || '—'}</div></div>
        <div class="result-item"><div class="result-label">Currency</div><div class="result-value">${d.currency_name || '—'} (${d.currency || '—'})</div></div>
        <div class="result-item"><div class="result-label">Calling Code</div><div class="result-value">+${d.country_calling_code ? d.country_calling_code.replace('+','') : '—'}</div></div>
      </div>
      ${mapUrl ? `<a class="map-link" href="${mapUrl}" target="_blank" rel="noopener">🗺 View on Map</a>` : ''}
    `;
  } catch(err) {
    content.innerHTML = `<div class="error-msg">⚠️ ${err.message}</div>`;
  }
}

function getFlagEmoji(code) {
  if (!code || code.length !== 2) return '🌐';
  return String.fromCodePoint(...[...code.toUpperCase()].map(c => 0x1F1E0 + c.charCodeAt(0) - 65));
}

fetchMyIP();

document.getElementById('ipInput').addEventListener('keydown', e => {
  if (e.key === 'Enter') lookupIP();
});
