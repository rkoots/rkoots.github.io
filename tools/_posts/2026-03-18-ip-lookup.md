<script>
let myIP = null;
let currentIPData = null;
let activeTab = 'overview';

// Fetch user's IP and display in hero section
async function fetchMyIP() {
  try {
    const res = await fetch('https://api.ipify.org?format=json');
    const data = await res.json();
    myIP = data.ip;
    document.getElementById('heroIp').textContent = myIP;
    
    // Detect IP version
    const isIPv6 = myIP.includes(':');
    document.getElementById('heroIpType').textContent = isIPv6 ? '✅ IPv6 Address' : '✅ IPv4 Address';
  } catch(e) {
    document.getElementById('heroIp').textContent = 'Unable to detect';
    document.getElementById('heroIpType').textContent = 'Connection error';
  }
}

// Copy hero IP to clipboard
function copyHeroIP() {
  if (myIP) {
    navigator.clipboard.writeText(myIP).then(() => {
      const btn = event.target;
      const originalText = btn.textContent;
      btn.textContent = '✅ Copied!';
      setTimeout(() => btn.textContent = originalText, 2000);
    }).catch(() => alert('IP: ' + myIP));
  }
}

// Show/hide IP guide
function showHideIPGuide() {
  const guide = document.getElementById('hideIPGuide');
  if (guide.style.display === 'none') {
    guide.style.display = 'block';
    guide.scrollIntoView({ behavior: 'smooth', block: 'start' });
  } else {
    guide.style.display = 'none';
  }
}

// Lookup user's own IP
async function lookupMyIP() {
  document.getElementById('ipInput').value = '';
  await lookupIP();
}

// Switch between tabs
function switchTab(tabName) {
  activeTab = tabName;
  
  // Update tab buttons
  document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active'));
  event.target.classList.add('active');
  
  // Update tab content
  document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));
  document.getElementById(tabName + 'Tab').classList.add('active');
}

// Main IP lookup function
async function lookupIP() {
  const ip = document.getElementById('ipInput').value.trim();
  const section = document.getElementById('resultSection');
  const overviewTab = document.getElementById('overviewTab');
  const locationTab = document.getElementById('locationTab');
  const securityTab = document.getElementById('securityTab');
  const networkTab = document.getElementById('networkTab');
  
  section.style.display = 'block';
  section.scrollIntoView({ behavior: 'smooth', block: 'start' });
  
  // Show loading state
  overviewTab.innerHTML = '<div class="loader"><div class="loader-spinner"></div><p>🔍 Fetching comprehensive IP data...</p></div>';
  locationTab.innerHTML = '<div class="loader"><div class="loader-spinner"></div></div>';
  securityTab.innerHTML = '<div class="loader"><div class="loader-spinner"></div></div>';
  networkTab.innerHTML = '<div class="loader"><div class="loader-spinner"></div></div>';

  try {
    const endpoint = ip ? `https://ipapi.co/${ip}/json/` : 'https://ipapi.co/json/';
    const res = await fetch(endpoint);
    if (!res.ok) throw new Error('API error ' + res.status);
    const d = await res.json();
    if (d.error) throw new Error(d.reason || 'Invalid IP address');
    
    currentIPData = d;
    
    // Populate all tabs
    populateOverviewTab(d);
    populateLocationTab(d);
    populateSecurityTab(d);
    populateNetworkTab(d);
    
  } catch(err) {
    const errorHtml = `<div class="error-msg">${err.message}</div>`;
    overviewTab.innerHTML = errorHtml;
    locationTab.innerHTML = errorHtml;
    securityTab.innerHTML = errorHtml;
    networkTab.innerHTML = errorHtml;
  }
}

// Populate Overview Tab

// Helper functions
function getFlagEmoji(code) {
  if (!code || code.length !== 2) return '🌐';
  return String.fromCodePoint(...[...code.toUpperCase()].map(c => 0x1F1E0 + c.charCodeAt(0) - 65));
}

function copyToClipboard(text) {
  navigator.clipboard.writeText(text).then(() => {
    const btn = event.target;
    const originalText = btn.textContent;
    btn.textContent = '✅';
    setTimeout(() => btn.textContent = originalText, 2000);
  }).catch(() => alert('Copied: ' + text));
}

function calculatePrivacyScore(isVPN, isProxy, isTor) {
  let score = 50; // Base score
  if (isVPN) score += 30;
  if (isProxy) score += 15;
  if (isTor) score += 40;
  return Math.min(score, 100);
}

function getScoreColor(score) {
  if (score >= 80) return '#10b981';
  if (score >= 50) return '#f59e0b';
  return '#ef4444';
}

function getScoreLabel(score) {
  if (score >= 80) return 'Excellent Privacy';
  if (score >= 50) return 'Moderate Privacy';
  return 'Low Privacy';
}

function getScoreDescription(score) {
  if (score >= 80) return 'Your connection has strong privacy protection with encryption and IP masking.';
  if (score >= 50) return 'Your connection has some privacy features but could be improved with a VPN.';
  return 'Your connection is exposed. Consider using a VPN for better privacy and security.';
}

function getPrivacyTip(isVPN, isProxy) {
  if (isVPN) return 'You\'re using a VPN, which provides good privacy protection. Make sure it\'s from a trusted provider.';
  if (isProxy) return 'Proxy servers provide basic IP masking but don\'t encrypt your traffic. Consider upgrading to a VPN.';
  return 'Your real IP address is exposed. Use a VPN service to encrypt your traffic and hide your IP for better privacy.';
}
