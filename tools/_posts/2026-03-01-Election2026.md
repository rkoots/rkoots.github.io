// Initialize default poll data on page load
function initializePollData() {
  Object.keys(STATE_PARTIES).forEach(state => {
    const stateRef = database.ref(`election-poll-2026/${state}`);
    stateRef.once('value').then((snapshot) => {
      if (!snapshot.exists()) {
        // Create default poll data for this state
        const defaultData = {};
        STATE_PARTIES[state].forEach(party => {
          defaultData[party.name] = 0;
        });
        defaultData.totalVotes = 0;
        stateRef.set(defaultData);
      }
    }).catch(error => {
      console.error(`Error initializing ${state}:`, error);
    });
  });
}

function displayLiveResults(data) {
  const container = document.getElementById('liveResultsContainer');
  const timestamp = new Date().toLocaleString('en-IN', { 
    timeZone: 'Asia/Kolkata',
    hour: '2-digit', 
    minute: '2-digit',
    day: 'numeric',
    month: 'short'
  });

  // Add dramatic loading effect
  container.innerHTML = '<div class="loading" style="padding: 40px;"><div style="font-size: 1.2rem; margin-bottom: 10px;">📊 Counting Votes...</div><div style="color: var(--muted);">Live results incoming...</div></div>';
  
  // Simulate loading delay for dramatic effect
  setTimeout(() => {
    let html = '<div class="live-results-grid">';
    
    Object.entries(data).forEach(([stateKey, stateData], stateIndex) => {
      const stateName = getStateDisplayName(stateKey);
      const isLeading = stateData.leading;
      
      // Sort parties by seats for dramatic reveal
      const sortedResults = [...stateData.results].sort((a, b) => b.seats - a.seats);
      
      html += `
        <div class="live-state-card ${isLeading ? 'leading' : ''}" style="animation-delay: ${stateIndex * 0.2}s">
          ${isLeading ? '<div class="live-leading-badge">🏆 LEADING</div>' : ''}
          <div class="live-state-header">
            <div class="live-state-name">${stateName}</div>
            <div class="live-seats-info">${stateData.countedSeats}/${stateData.totalSeats} seats counted</div>
          </div>
          
          <div class="live-party-results">
      `;
      
      sortedResults.forEach((party, partyIndex) => {
        const isWinner = partyIndex === 0 && party.seats > 0;
        html += `
          <div class="live-party-result" style="animation-delay: ${stateIndex * 0.2 + partyIndex * 0.1}s">
            <div class="live-party-info">
              <div class="live-party-badge" style="background-color: ${party.color}"></div>
              <div class="live-party-details">
                <div class="live-party-name">${party.party} ${isWinner ? '👑' : ''}</div>
                <div class="live-party-seats">${party.seats} seat${party.seats !== 1 ? 's' : ''}</div>
              </div>
            </div>
            <div class="live-seats-count" data-count="${party.seats}">0</div>
          </div>
        `;
      });
      
      html += `
          </div>
        </div>
      `;
    });
    
    html += '</div>';
    html += `
      <div class="live-timestamp" style="animation: fadeIn 1s ease-out">
        <span class="live-updating"></span>
        Last updated: ${timestamp} IST | Results updated every 30 seconds
      </div>
    `;
    
    container.innerHTML = html;
    
    // Animate seat counting
    setTimeout(() => {
      document.querySelectorAll('.live-seats-count').forEach(counter => {
        const targetCount = parseInt(counter.getAttribute('data-count'));
        animateCounter(counter, 0, targetCount, 1500);
      });
    }, 500);
    
    // Add celebration effect for leading parties
    setTimeout(() => {
      document.querySelectorAll('.live-state-card.leading').forEach(card => {
        card.style.animation = 'bounce 0.6s ease-out, glow 2s infinite';
      });
    }, 1000);
    
  }, 800);
}

function animateCounter(element, start, end, duration) {
  const startTime = performance.now();
  const isNegative = end < 0;
  end = Math.abs(end);
  
  function update(currentTime) {
    const elapsed = currentTime - startTime;
    const progress = Math.min(elapsed / duration, 1);
    
    // Easing function for smooth animation
    const easeOutQuart = 1 - Math.pow(1 - progress, 4);
    const current = Math.floor(start + (end - start) * easeOutQuart);
    
    element.textContent = (isNegative ? -current : current);
    
    if (progress < 1) {
      requestAnimationFrame(update);
    }
  }
  
  requestAnimationFrame(update);
}

function getStateDisplayName(stateKey) {
  const stateNames = {
    'assam': 'Assam',
    'kerala': 'Kerala',
    'puducherry': 'Puducherry',
    'tamil-nadu': 'Tamil Nadu',
    'west-bengal': 'West Bengal'
  };
  return stateNames[stateKey] || stateKey;
}

// FAQ Toggle Function
function toggleFAQ(element) {
  const faqItem = element.parentElement;
  const isActive = faqItem.classList.contains('active');
  
  // Close all FAQ items
  document.querySelectorAll('.faq-item').forEach(item => {
    item.classList.remove('active');
  });
  
  // Open clicked item if it wasn't active
  if (!isActive) {
    faqItem.classList.add('active');
  }
}

// FAQ Search Function
function filterFAQs() {
  const searchInput = document.getElementById('faqSearch');
  const filter = searchInput.value.toLowerCase();
  const faqItems = document.querySelectorAll('.faq-item');
  let visibleCount = 0;
  
  faqItems.forEach(item => {
    const question = item.querySelector('.faq-question span').textContent.toLowerCase();
    const answer = item.querySelector('.faq-answer p').textContent.toLowerCase();
    const tags = item.getAttribute('data-tags').toLowerCase();
    
    if (question.includes(filter) || answer.includes(filter) || tags.includes(filter)) {
      item.style.display = 'block';
      visibleCount++;
    } else {
      item.style.display = 'none';
    }
  });
  
  // Update FAQ count
  document.getElementById('faqCount').textContent = visibleCount;
  
  // Show no results message if needed
  if (visibleCount === 0 && filter.length > 0) {
    if (!document.querySelector('.no-results')) {
      const noResults = document.createElement('div');
      noResults.className = 'no-results';
      noResults.style.cssText = 'text-align: center; padding: 20px; color: var(--muted);';
      noResults.innerHTML = `<p>😔 No FAQs found matching "${filter}"</p><p>Try different keywords or browse all FAQs above.</p>`;
      document.querySelector('.faq-container').appendChild(noResults);
    }
  } else {
    const noResults = document.querySelector('.no-results');
    if (noResults) {
      noResults.remove();
    }
  }
}
