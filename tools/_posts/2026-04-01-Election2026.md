// Initialize default poll data on page load
function initializePollData() {
  Object.keys(STATE_PARTIES).forEach(state => {
    const stateRef = database.ref(`elections/2026/${state}`);
    stateRef.once('value').then((snapshot) => {
      if (!snapshot.exists()) {
        // Create default poll data for this state
        const defaultData = {};
        STATE_PARTIES[state].forEach(party => {
          defaultData[party.name] = 0;
        });
        defaultData.totalVotes = 0;
        stateRef.set(defaultData);
      }
    }).catch(error => {
      console.error(`Error initializing ${state}:`, error);
    });
  });
}

function loadStatePoll() {
  const stateSelect = document.getElementById('stateSelect');
  const pollContainer = document.getElementById('pollContainer');
  
  currentState = stateSelect.value;
  selectedParty = '';
  hasVoted = false;
  
  if (!currentState) {
    pollContainer.classList.remove('active');
    return;
  }
  
  pollContainer.classList.add('active');
  document.getElementById('loadingMessage').style.display = 'block';
  document.getElementById('votingSection').style.display = 'none';
  document.getElementById('resultsSection').style.display = 'none';
  document.getElementById('successMessage').style.display = 'none';
  
  // Load current poll data from Firebase
  const stateRef = database.ref(`elections/2026/${currentState}`);
  stateRef.on('value', (snapshot) => {
    const data = snapshot.val() || {};
    displayParties();
    displayResults(data);
    document.getElementById('loadingMessage').style.display = 'none';
    document.getElementById('votingSection').style.display = 'block';
    document.getElementById('resultsSection').style.display = 'block';
  });
}

function displayParties() {
  const partiesGrid = document.getElementById('partiesGrid');
  const parties = STATE_PARTIES[currentState];

      <div class="party-name">${party.name}</div>
      <div style="font-size: 0.85rem; color: var(--muted);">${party.fullName}</div>
    </div>
  `).join('');
}

function selectParty(partyName) {
  if (hasVoted) return;
  
  selectedParty = partyName;
  
  // Update UI
  document.querySelectorAll('.party-card').forEach(card => {
    card.classList.remove('selected');
  });
  document.querySelector(`[data-party="${partyName}"]`).classList.add('selected');
  
  // Enable submit button
  document.getElementById('submitVote').disabled = false;
}

function submitVote() {
  if (!selectedParty || hasVoted) return;
  
  const stateRef = database.ref(`elections/2026/${currentState}`);
  
  // Update vote count
  stateRef.transaction((currentData) => {
    if (!currentData) {
      currentData = {};
      STATE_PARTIES[currentState].forEach(party => {
        currentData[party.name] = 0;
      });
      currentData.totalVotes = 0;
    }
    
    currentData[selectedParty] = (currentData[selectedParty] || 0) + 1;
    currentData.totalVotes = (currentData.totalVotes || 0) + 1;
    
    return currentData;
  }).then(() => {
    hasVoted = true;
    selectedParty = '';
    document.getElementById('submitVote').disabled = true;
    document.getElementById('successMessage').style.display = 'block';
    
    // Clear selection
    document.querySelectorAll('.party-card').forEach(card => {
      card.classList.remove('selected');
    });
    
    // Hide success message after 3 seconds
    setTimeout(() => {
      document.getElementById('successMessage').style.display = 'none';
    }, 3000);
  }).catch(error => {
    console.error('Error submitting vote:', error);
    alert('Error submitting vote. Please try again.');
  });
}

function displayResults(data) {
  const resultsGrid = document.getElementById('resultsGrid');
  const parties = STATE_PARTIES[currentState];
  const totalVotes = data.totalVotes || 0;
  
  if (totalVotes === 0) {
    resultsGrid.innerHTML = '<p style="text-align: center; color: var(--muted);">No votes recorded yet. Be the first to vote!</p>';
    return;
  }
  
  resultsGrid.innerHTML = parties.map(party => {
    const votes = data[party.name] || 0;
    const percentage = totalVotes > 0 ? ((votes / totalVotes) * 100).toFixed(1) : 0;
    
    return `
      <div class="result-item">
        <div class="result-party">
          <div class="party-color" style="background-color: ${party.color}"></div>
          <div>
            <div style="font-weight: 600;">${party.name}</div>
            <div style="font-size: 0.85rem; color: var(--muted);">${party.fullName}</div>
          </div>
        </div>
        <div class="result-bar">
          <div class="result-fill" style="width: ${percentage}%"></div>
        </div>
        <div class="result-votes">${votes} (${percentage}%)</div>
      </div>
    `;
  }).join('');
}
