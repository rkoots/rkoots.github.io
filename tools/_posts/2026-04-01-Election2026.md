// State selection function for widget
function selectState(stateKey) {
  // Update UI
  document.querySelectorAll('.state-card').forEach(card => {
    card.classList.remove('selected');
  });
  document.querySelector(`[data-state="${stateKey}"]`).classList.add('selected');

  loadStatePoll();

// Initialize default poll data on page load
function initializePollData() {
  Object.keys(STATE_PARTIES).forEach(state => {
    const stateRef = database.ref(`election-poll-2026/${state}`);
    stateRef.once('value').then((snapshot) => {
      if (!snapshot.exists()) {
        // Create default poll data for this state
        const defaultData = {};
        STATE_PARTIES[state].forEach(party => {
          defaultData[party.name] = 0;
        });
        defaultData.totalVotes = 0;
        stateRef.set(defaultData);
      }
    }).catch(error => {
      console.error(`Error initializing ${state}:`, error);
    });
  });
}

function loadStatePoll() {

  selectedParty = '';

  if (!currentState) {
    pollContainer.classList.remove('active');
    return;

  document.getElementById('votingSection').style.display = 'none';
  document.getElementById('resultsSection').style.display = 'none';

}

      <div class="party-name">${party.name}</div>
      <div style="font-size: 0.85rem; color: var(--muted);">${party.fullName}</div>
    </div>
  `).join('');
}

function selectParty(partyName) {
  if (hasVoted) return;
  
  selectedParty = partyName;
  
  // Update UI
  document.querySelectorAll('.party-card').forEach(card => {
    card.classList.remove('selected');
  });
  document.querySelector(`[data-party="${partyName}"]`).classList.add('selected');
  
  // Enable submit button
  document.getElementById('submitVote').disabled = false;
}

function submitVote() {
  if (!selectedParty || hasVoted) return;
  
  const submitBtn = document.getElementById('submitVote');
  
  // Add submit animation class
  submitBtn.classList.add('submit-animation');
  
  // Disable button during submission
  submitBtn.disabled = true;
  submitBtn.innerHTML = '⏳ Submitting...';
  
  const stateRef = database.ref(`election-poll-2026/${currentState}`);
  
  // Update vote count
  stateRef.transaction((currentData) => {
    if (!currentData) {
      currentData = {};
      STATE_PARTIES[currentState].forEach(party => {
        currentData[party.name] = 0;
      });
      currentData.totalVotes = 0;
    }
    
    currentData[selectedParty] = (currentData[selectedParty] || 0) + 1;
    currentData.totalVotes = (currentData.totalVotes || 0) + 1;
    
    return currentData;
  }).then(() => {
    // Show success animation
    showVoteSuccessAnimation();
    
    // Update state
    hasVoted = true;
    selectedParty = '';
    
    // Reset button after delay
    setTimeout(() => {
      submitBtn.disabled = true;
      submitBtn.innerHTML = '✅ Vote Submitted';
      submitBtn.classList.remove('submit-animation');
    }, 2000);
    
    // Clear selection
    document.querySelectorAll('.party-card').forEach(card => {
      card.classList.remove('selected');
    });
    
    // Hide success message after delay
    setTimeout(() => {
      document.getElementById('successMessage').style.display = 'none';
    }, 3000);
  }).catch(error => {
    console.error('Error submitting vote:', error);
    submitBtn.disabled = false;
    submitBtn.innerHTML = '🗳️ Submit Vote';
    submitBtn.classList.remove('submit-animation');
    alert('Error submitting vote. Please try again.');
  });
}

function createConfetti() {

  container.className = 'confetti-container';

    confetti.className = 'confetti';
    confetti.style.left = Math.random() * 100 + '%';
    confetti.style.animationDelay = Math.random() * 0.5 + 's';
    confetti.style.animationDuration = (2 + Math.random() * 2) + 's';
    container.appendChild(confetti);
  }
  
  document.body.appendChild(container);

    container.remove();

}

function displayResults(data) {
  const resultsGrid = document.getElementById('resultsGrid');
  const parties = STATE_PARTIES[currentState];
  const totalVotes = data.totalVotes || 0;
  
  if (totalVotes === 0) {
    resultsGrid.innerHTML = '<p style="text-align: center; color: var(--muted); animation: fadeIn 0.5s ease-out;">No votes recorded yet. Be the first to vote!</p>';
    return;
  }

  const sortedParties = parties.map(party => ({
    ...party,

    percentage: totalVotes > 0 ? ((data[party.name] || 0) / totalVotes * 100).toFixed(1) : 0

        <div class="party-color" style="background-color: ${party.color}"></div>

      </div>
      <div class="result-votes">${party.votes} (${party.percentage}%)</div>

  setTimeout(() => {
    document.querySelectorAll('.result-fill').forEach(bar => {
      bar.style.width = bar.getAttribute('data-width');
    });
  }, 100);
}

  const timestamp = new Date().toLocaleString('en-IN', {

    hour: '2-digit', 
    minute: '2-digit',
    day: 'numeric',

function animateCounter(element, start, end, duration) {
  const startTime = performance.now();
  const isNegative = end < 0;
  end = Math.abs(end);
  
  function update(currentTime) {
    const elapsed = currentTime - startTime;
    const progress = Math.min(elapsed / duration, 1);
    
    // Easing function for smooth animation
    const easeOutQuart = 1 - Math.pow(1 - progress, 4);
    const current = Math.floor(start + (end - start) * easeOutQuart);
    
    element.textContent = (isNegative ? -current : current);
    
    if (progress < 1) {
      requestAnimationFrame(update);
    }
  }
  
  requestAnimationFrame(update);
}

  };
  return stateNames[stateKey] || stateKey;
}
